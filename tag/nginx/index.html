    常用設定

        # 設定監聽的 port
    listen          80;
    # 定義伺服器名稱
    server_name     localhost;
    # 設定專案資料夾
    root            /home/tca/vue-project/dist;

    index           index.html;

    charset utf-8;
    # 隱藏伺服器版本資訊
    server_tokens off;

    form-action 'self';
    frame-src 'none';
    frame-ancestors 'none';
    " always;
