                                    # 環境安裝

                                    • windows 安裝：參閱 [[install_windows_laravel]] 內的說明
                                    • ubuntu 安裝：參閱 [[install_ubuntu_laravel]] 內的說明

                                    # 專案安裝

                                    # 初次啟動專案需要先下載相依套件

                                    composer install
npm install

                                    # 還原.env 設定檔，需要根據新環境去調整資料庫連線設定

                                    cp .env.apluspms .env
php artisan key:generate

                                    # 重建資料庫

                                    php artisan migrate

                                    # 重建資料庫，若專案有指定路徑

                                    php artisan migrate --path=/database/migrations/apluspms

                                    # 啟動框架的伺服器

                                    npm run dev
php artisan serv

                                    # 打包前端資源

                                    npm run build
